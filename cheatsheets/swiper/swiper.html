import 'swiper/css';

🎬 Recipe：基础轮播怎么配 >>>

最常见的是“一个横向轮播 + 分页 + 响应式断点 + 轻微过渡”。这种场景不要上复杂特效，优先把滑动体验调顺。
  • slidesPerView：每屏显示几张
  • spaceBetween：卡片间距
  • breakpoints：按屏宽切换布局
  • centeredSlides：居中展示首屏
  • watchOverflow：滑块太少时自动禁用交互
  • grabCursor：桌面端更像可拖拽组件
const swiper = new Swiper('.swiper', {
  slidesPerView: 1.15,
  spaceBetween: 12,
  centeredSlides: false,
  grabCursor: true,
  watchOverflow: true,
  breakpoints: {
    640: { slidesPerView: 2, spaceBetween: 16 },
    1024: { slidesPerView: 3, spaceBetween: 20 },
  },
});

🖼️ Recipe：缩略图联动大图 >>>

图库、商品详情、作品集最常见的组合是“主图 + 缩略图”。关键不是事件，而是 `Thumbs` 和 `watchSlidesProgress`。
  • thumbs: { swiper: thumbsSwiper }
  • watchSlidesProgress: true
  • 缩略图通常关掉复杂交互，只保留点击切换
  • loop watchOverflow
import Swiper from 'swiper';
import { Thumbs, FreeMode } from 'swiper/modules';
import 'swiper/css';

const thumbs = new Swiper('.thumbs', {
  modules: [FreeMode],
  slidesPerView: 5,
  spaceBetween: 8,
  freeMode: true,
  watchSlidesProgress: true,
});

const main = new Swiper('.main', {
  modules: [Thumbs],
  slidesPerView: 1,
  spaceBetween: 12,
  thumbs: {
    swiper: thumbs,
  },
});

🧮 Recipe：海量数据与虚拟列表 >>>

当 slide 很多时，重点是减少 DOM 压力，而不是一味调动画。虚拟列表适合长列表、日志页、卡片墙和动态数据源。
  • virtual：开启虚拟滑块
  • renderExternal：把渲染交给外部框架
  • renderExternalUpdate：异步渲染后手动更新
  • slidesPerView: 'auto' slidesPerViewAutoSlideSize
  • update()
import Swiper from 'swiper';
import { Virtual } from 'swiper/modules';

const swiper = new Swiper('.swiper', {
  modules: [Virtual],
  virtual: {
    slides: Array.from({ length: 1000 }, (_, i) => `Slide ${i + 1}`),
  },
  slidesPerView: 3,
  spaceBetween: 12,
});
swiper.virtual.slides.push('New slide');
swiper.update();

✨ Recipe：特效和动效 >>>

Swiper 的特效模块适合做封面、展示页和轻量营销页。不要在复杂业务列表里乱堆特效，维护成本会很快上来。
  • fade cube coverflow flip cards creative：、、、、、
  • swiper/css/effect-fade
  • fade creative cards
import Swiper from 'swiper';
import { EffectFade, EffectCreative } from 'swiper/modules';

import 'swiper/css';
import 'swiper/css/effect-fade';
import 'swiper/css/effect-creative';

const fadeSwiper = new Swiper('.fade-swiper', {
  modules: [EffectFade],
  effect: 'fade',
  fadeEffect: {
    crossFade: true,
  },
});

const creativeSwiper = new Swiper('.creative-swiper', {
  modules: [EffectCreative],
  effect: 'creative',
  creativeEffect: {
    prev: {
      shadow: true,
      translate: ['-20%', 0, -1],
    },
    next: {
      translate: ['100%', 0, 0],
    },
  },
});

🧱 Recipe：React / Vue / Web Components >>>

Swiper 的三种官方入口共享同一套核心参数，但组件封装层不同。React / Vue 用组件，Web Components 用自定义元素。
  • swiper/react
  • swiper/vue
  • swiper/element swiper/element/bundle：或
  • modules
  • swiper/element register()
// React
import { Swiper, SwiperSlide } from 'swiper/react';
import { Navigation, Pagination } from 'swiper/modules';
import 'swiper/css';

export default function App() {
  return (
    <Swiper
      modules={[Navigation, Pagination]}
      spaceBetween={24}
      slidesPerView={3}
      navigation
      pagination={{ clickable: true }}
      onSlideChange={() => console.log('slide change')}
    >
      <SwiperSlide>Slide 1</SwiperSlide>
      <SwiperSlide>Slide 2</SwiperSlide>
      <SwiperSlide>Slide 3</SwiperSlide>
    </Swiper>
  );
}
<!-- Vue -->
<template>
  <Swiper
    :modules="[Navigation, Pagination]"
    :slides-per-view="3"
    :space-between="24"
    navigation
    :pagination="{ clickable: true }"
    @slideChange="onSlideChange"
  >
    <SwiperSlide>Slide 1</SwiperSlide>
    <SwiperSlide>Slide 2</SwiperSlide>
    <SwiperSlide>Slide 3</SwiperSlide>
  </Swiper>
</template>

<script setup>
import { Swiper, SwiperSlide } from 'swiper/vue';
import { Navigation, Pagination } from 'swiper/modules';
import 'swiper/css';

const onSlideChange = () => {
  console.log('slide change');
};
</script>
<!-- Web Components -->
<swiper-container init="false"></swiper-container>

<script type="module">
  import { register } from 'swiper/element/bundle';

  register();

  const el = document.querySelector('swiper-container');
  Object.assign(el, {
    slidesPerView: 3,
    navigation: true,
    pagination: { clickable: true },
  });
  el.initialize();
</script>

📚 Quick Ref：模块、方法、事件 >>>

模块决定“能做什么”，方法决定“怎么驱动”，事件决定“什么时候接业务逻辑”。

Thumbs缩略图联动swiper

⚠️ 常见坑 / 决策规则 >>>

Swiper 的坑通常来自“模块没引、样式没引、结构没对、模式没选对”。先排这些基础项，再看业务逻辑。
  • 只写 JS 不引 CSS，样式会直接不完整
  • React / Vue / Element 都不是同一个入口，别混着用
  • thumbs watchSlidesProgress
  • loop rewind
  • nextEl prevEl el
  • virtual update()
  • cssMode
  • loading="lazy"
