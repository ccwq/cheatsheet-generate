🗺️ 地图初始化与配置 >>>

  • L.map('mapId', options) ：创建地图实例
  • center: [lat, lng] ：设置地图中心点
  • zoom: 10 ：设置初始缩放级别
  • minZoom: 1, maxZoom: 18 ：缩放范围限制
  • maxBounds: bounds ：地图边界限制
  • zoomControl: true ：显示缩放控件
  • attributionControl: true ：显示版权控件

交互控制

  • dragging: true ：启用拖拽
  • touchZoom: true ：触摸缩放
  • scrollWheelZoom: true ：滚轮缩放
  • doubleClickZoom: true ：双击缩放
  • keyboard: true ：键盘控制

地图初始化示例

let map = L.map('map', {
    center: [39.9042, 116.4074],
    zoom: 12,
    zoomControl: true,
    scrollWheelZoom: true
});
基础地图配置，支持多种交互选项

🎯 视图控制 >>>

  • map.setView([lat, lng], zoom) ：设置视图
  • map.setZoom(zoom) ：设置缩放级别
  • map.zoomIn(delta) ：放大
  • map.zoomOut(delta) ：缩小
  • map.panTo([lat, lng]) ：平移到指定点
  • map.flyTo([lat, lng], zoom) ：飞行动画到指定点
  • map.fitBounds(bounds) ：适应边界
  • map.setMaxBounds(bounds) ：设置最大边界

动画选项

map.flyTo([39.9042, 116.4074], 15, {
    duration: 2,
    easeLinearity: 0.5
});

边界操作

let bounds = L.latLngBounds(
    [[39.8, 116.3], [40.0, 116.5]]

map.fitBounds(bounds, {padding: [20, 20]});
支持平滑动画和边界自适应

🗂️ 瓦片图层系统 >>>

  • L.tileLayer(url, options) ：创建瓦片图层
  • url: 'https://{s}.domain.com/{z}/{x}/{y}.png' ：瓦片URL模板
  • attribution: '© attribution' ：版权信息
  • maxZoom: 19 ：最大缩放级别
  • subdomains: ['a','b','c'] ：子域名
  • opacity: 1.0 ：图层透明度
  • zIndex: 1 ：图层堆叠顺序

常用瓦片源

// OpenStreetMap
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png')

// 高德地图
L.tileLayer('https://webrd0{s}.is.autonavi.com/appmaptile?lang=zh_cn&size=1&scale=1&style=7&x={x}&y={y}&z={z}')

// 腾讯地图
L.tileLayer('https://rt{s}.map.gtimg.com/realtimerender?z={z}&x={x}&y={y}&type=vector&style=0')

WMS图层

L.tileLayer.wms('http://wms-server/wms', {
    layers: 'layer_name',
    format: 'image/png',
    transparent: true,
    version: '1.1.1'

  • pointToLayer: function ：点转换函数

样式设置

L.geoJSON(data, {
    style: function(feature) {
        return {
            color: feature.properties.color || '#3388ff',
            weight: 2,
            fillOpacity: 0.6
        };
    }
}).addTo(map);

点标记转换

L.geoJSON(data, {
    pointToLayer: function(feature, latlng) {
        return L.circleMarker(latlng, {
            radius: 8,
            fillColor: "#ff7800",
            color: "#000",
            weight: 1,
            opacity: 1,
            fillOpacity: 0.8
        });
    }
}).addTo(map);

要素处理

L.geoJSON(data, {
    onEachFeature: function(feature, layer) {
        if (feature.properties && feature.properties.name) {
            layer.bindPopup(feature.properties.name);
        }
    }
}).addTo(map);

⚡ 性能优化技巧 >>>

  • updateWhenIdle: false ：实时更新

    preferCanvas: true,
    renderer: L.canvas()

分批加载

function addMarkersInBatches(markers, batchSize = 100) {
    let index = 0;

    function addBatch() {
        for (let i = 0; i < batchSize && index < markers.length; i++, index++) {
            markers[index].addTo(map);
        }

        if (index < markers.length) {
            setTimeout(addBatch, 10);
        }
    }

    addBatch();
}

视图范围管理

map.on('moveend', function() {
    let bounds = map.getBounds();

    allMarkers.forEach(marker => {
        if (bounds.contains(marker.getLatLng())) {
            if (!map.hasLayer(marker)) {
                marker.addTo(map);
            }
        } else {
            if (map.hasLayer(marker)) {
                map.removeLayer(marker);
            }
        }
    });
});
